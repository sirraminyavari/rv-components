<IconGallery
  style={{
    display: 'grid',
    gridTemplateColumns: 'repeat(auto-fill,minmax(250px,1fr))',
  }}
>
  {Object.entries(icons).map(([name, Icon]) => {
    if (name.startsWith('CM')) return;
    if (['CaretSvg', 'ChevronSvg', 'ArrowCircleSvg'].includes(name))
      return ['right', 'up', 'left', 'down'].map((direction) => (
        <Fragment key={name + direction}>
          <IconItem name={`${name.replace('Svg', '')}-${direction}`}>
            <Icon
              direction={direction}
              name={`${name.replace('Svg', '')}-${direction}`}
            />
          </IconItem>
          <IconItem name={`${name.replace('Svg', '')}-${direction} (outline)`}>
            <Icon
              direction={direction}
              name={`${name.replace('Svg', '')}-${direction} (outline)`}
              outline
            />
          </IconItem>
        </Fragment>
      ));
    return (
      <Fragment key={name}>
        <IconItem name={name.replace('Svg', '')}>
          <Icon name={name.replace('Svg', '')} />
        </IconItem>
        <IconItem name={`${name.replace('Svg', '')} (outline)`}>
          <Icon name={`${name.replace('Svg', '')} (outline)`} outline />
        </IconItem>
      </Fragment>
    );
  })}
</IconGallery>
